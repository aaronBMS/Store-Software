    NOMBRE VARCHAR(30) NOT NULL,
    APELLIDO VARCHAR(30) NOT NULL,
    TELEFONO INT NOT NULL,
    CORREO VARCHAR(30) NOT NULL,
    USUARIO VARCHAR(20) UNIQUE NOT NULL,
    CONTRASEÃ‘A VARCHAR(20) NOT NULL,
    CONDICION VARCHAR(20) NOT NULL
);

CREATE TABLE PROVEEDOR(
    RUC INT PRIMARY KEY,
    NOMBRE VARCHAR(30) NOT NULL,
    DIRECCION VARCHAR(50) NOT NULL,
    CIUDAD VARCHAR(30) NOT NULL,
    CORREO VARCHAR(30) NOT NULL,
    TELEFONO INT NOT NULL,
    OBSERVACIONES VARCHAR(60)
);  

CREATE TABLE PRODUCTO(
    CODIGO INT PRIMARY KEY,
    NOMBRE VARCHAR(30) NOT NULL,
    CATEGORIA VARCHAR(30) NOT NULL,
    MARCA VARCHAR(30) NOT NULL,
    STOCK_INICIAL INT NOT NULL,
    STOCK_MINIMO INT NOT NULL,
    PRECIO_COSTO DOUBLE(6,2) NOT NULL,
    PRECIO_VENTA DOUBLE(6,2) NOT NULL,
    UTILIDAD DOUBLE(6,2) NOT NULL,
    IMAGEN BLOB NOT NULL
);

CREATE TABLE CLIENTE(
    CODIGO INT NOT NULL,
    NOMBRE VARCHAR(30) NOT NULL,
    APELLIDO VARCHAR(30) NOT NULL,
    TELEFONO INT NOT NULL,
    CORREO VARCHAR(30)
);

CREATE TABLE VENTA(
    CODIGO INT PRIMARY KEY,
    CODIGO_CLIENTE INT NOT NULL,
    CLIENTE VARCHAR(30) NOT NULL,
    RESPONSABLE VARCHAR(30) NOT NULL,
    FECHA VARCHAR(30) NOT NULL,
    OBSERVACIONES VARCHAR(30),
    PAGO_TOTAL DOUBLE(6,2) NOT NULL
);

CREATE TABLE DETALLE_VENTA(
    CODIGO_VENTA INT PRIMARY KEY,
    CODIGO_PRODUCTO INT NOT NULL,
    NOMBRE VARCHAR(30) NOT NULL,
    CANTIDAD INT NOT NULL,
    PRECIO DOUBLE(6,2) NOT NULL,
    DESCUENTO DOUBLE(6,2),
    TOTOAL DOUBLE(6,2) NOT NULL
);

CREATE TABLE COMPRA(
    CODIGO INT PRIMARY KEY,
    RUC INT NOT NULL,
    PROVEEDOR VARCHAR(30) NOT NULL,
    RESPONSABLE VARCHAR(30) NOT NULL,
    FECHA VARCHAR(30) NOT NULL,
    OBSERVACIONES VARCHAR(30),
    PAGO_TOTAL DOUBLE(6,2) NOT NULL
);

CREATE TABLE DETALLE_COMPRA(
    CODIGO_COMPRA INT PRIMARY KEY,
    CODIGO_PRODUCTO INT NOT NULL,
    NOMBRE VARCHAR(30) NOT NULL,
    CANTIDAD INT NOT NULL,
    PRECIO DOUBLE(6,2) NOT NULL,
    DESCUENTO DOUBLE(6,2),
    TOTAL DOUBLE(6,2) NOT NULL
);
